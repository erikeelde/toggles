Quick Start with Toggles-Flow

The recommended reactive approach using Kotlin Flow

// Add to your build.gradle
implementation("se.eelde.toggles:toggles-flow:0.0.3")

// Use in your Activity/Fragment
import se.eelde.toggles.flow.Toggles

val toggles = Toggles(this)

lifecycleScope.launch {
    toggles.toggle("enable_new_feature", false).collect { isEnabled ->
        if (isEnabled) {
            showNewFeature()
        } else {
            showOldFeature()
        }
    }
}

Support Multiple Data Types

// Boolean toggles
toggles.toggle("enable_feature", false)

// String configurations  
toggles.toggle("api_endpoint", "https://api.example.com")

// Integer values
toggles.toggle("max_retry_count", 3)

// Enum configurations
toggles.toggle("log_level", LogLevel::class.java, LogLevel.INFO)
